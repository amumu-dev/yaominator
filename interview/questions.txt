isMirror( Node leftroot , Node rightroot )
{
    if( leftroot == null && rightroot == null )
        return true;

    if( leftroot.data == rightroot.data )
    {
        return isMirror( leftroot.left , rightroot.right ) && isMirror( leftroot.right , rightroot.left);

    }
    else
        return false;
    
}
